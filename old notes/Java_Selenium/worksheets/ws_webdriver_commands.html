Task 01: Open URL and Get Page Info
Use driver.get() to open "https://opensource-demo.orangehrmlive.com/". Print the page title using getTitle() and current URL using getCurrentUrl(). Verify the title equals "OrangeHRM".
Task 02: Enter Text in Input Fields
Locate the username field by name "username" and use sendKeys() to enter "Admin". Locate the password field by name "password" and enter "admin123". Print confirmation messages after each entry.
(Instruction: Use the ID locators 'user' and 'pass' for these fields)
Task 03: Click Button and Maximize Window
Use driver.manage().window().maximize() to maximize the browser window. Locate the login button and click it using click(). Get the page source length using getPageSource().length() and print it.

package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class _04_Webdrivercommands {

    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        try {
            // Launch Chrome
            WebDriver driver =
                new ChromeDriver();

            System.out.println("Navigating to WebDriver Commands worksheet...");
            // Open worksheet page
            driver.get(
                "http://127.0.0.1:5501/Java_Selinium/worksheets/ws_webdriver_commands.html"
            );

            driver.manage()
                .window()
                .maximize();

            // Task 01: Open URL and Get Page Info
            System.out.println("Executing Task 01: Loading demo site and fetching title/URL...");
            driver.get(
                "https://opensource-demo.orangehrmlive.com/"
            );
            System.out.println(
                "Page Title: " + driver.getTitle()
            );
            System.out.println(
                "Current URL: " + driver.getCurrentUrl()
            );

            Thread.sleep(1000);

            // Task 02: Interaction (Using IDs as per instruction)
            System.out.println("Executing Task 02: Entering credentials using IDs...");
            System.out.println("Entering Username: Admin");
            driver.findElement(By.id("user")).sendKeys("Admin");
            System.out.println("Entering Password: admin123");
            driver.findElement(By.id("pass")).sendKeys("admin123");

            Thread.sleep(1000);

            // Task 03: Click & Source
            System.out.println("Executing Task 03: Clicking Login and checking page source...");
            driver.findElement(By.id("loginBtn")).click();

            System.out.println(
                "Source Length: " +
                driver.getPageSource()
                    .length()
            );

            System.out.println("--- Selenium Lab Session Ended ---");
            driver.quit();

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
