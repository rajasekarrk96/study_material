Task 01: Drag and Drop Method
Use Actions.dragAndDrop(source, target) to move "Task A" (id="task1") from the To-Do column to the Done column (id="col-done").
(Instruction: Use the ID locators 'task1' and 'col-done' for this task)
Task 02: Click, Hold, Move, Release
Use clickAndHold(), moveToElement(), and release() to drag "Task B" to the Done column step by step.

package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.interactions.Actions;

public class _17_Draganddrop {
    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver = new ChromeDriver();

        System.out.println("Navigating to Drag & Drop worksheet...");
        driver.get("http://127.0.0.1:5501/Java_Selinium/worksheets/ws_drag_drop.html");

        Actions act = new Actions(driver);

        // Task 01: Drag & Drop using IDs
        System.out.println("Executing Task 01: Dragging 'Task A' to 'Done' column...");
        WebElement source = driver.findElement(By.id("task1"));
        WebElement target = driver.findElement(By.id("col-done"));
        act.dragAndDrop(source, target).perform();
        System.out.println("'Task A' moved to 'Done' column.");

        // Task 02: Click, Hold, Move, Release
        System.out.println("Executing Task 02: Manually dragging 'Task B' to 'Done' column...");
        WebElement taskB = driver.findElement(By.xpath("//div[text()='Task B']"));
        act.clickAndHold(taskB)
           .moveToElement(target)
           .release()
           .perform();
        System.out.println("'Task B' moved to 'Done' column successfully.");

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
