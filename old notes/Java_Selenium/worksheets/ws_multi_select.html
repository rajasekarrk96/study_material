package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.Select;

public class _21_Multiselectdropdown {
    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver =
        new ChromeDriver();

        System.out.println("Navigating to Multi-Select worksheet...");
        driver.get(
            "http://127.0.0.1:5501/Java_Selinium/worksheets/ws_multi_select.html"
        );

        // Task 01–03: Multi-Select Dropdown
        System.out.println("Executing Task 1-3: Working with Multi-Select Dropdown...");
        Select select =
        new Select(
            driver.findElement(
                By.id("fruits")
            )
        );

        // Task 01: Check if dropdown is multi-select
        boolean isMulti =
        select.isMultiple();

        System.out.println("Dropdown ID: 'fruits'");
        System.out.println(
            "Is Multi-Select: " + isMulti
        );

        // Task 02: Select multiple options
        if(isMulti) {
            System.out.println("Selecting 'Apple'...");
            select.selectByVisibleText("Apple");
            System.out.println("Selecting 'Banana'...");
            select.selectByVisibleText("Banana");
        }

        // Task 03: Deselect all options
        System.out.println("Deselecting all options...");
        select.deselectAll();

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
