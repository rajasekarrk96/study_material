package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class _08_XPath {

    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");

        // Initialize ChromeDriver
        WebDriver driver = new ChromeDriver();
        driver.manage().window().maximize();
        
        System.out.println("Navigating to XPath worksheet...");
        driver.get("http://127.0.0.1:5501/Java_Selinium/worksheets/ws_xpath.html");

        // Task 01: Attribute Match
        System.out.println("Executing Task 01: Finding Dashboard Title via class attribute...");
        driver.findElement(By.xpath("//h2[@class='dashboard-title']"));
        System.out.println("Found Dashboard title.");

        // Task 02: Exact Text Match
        System.out.println("Executing Task 02: Finding Log Out link via exact text...");
        driver.findElement(By.xpath("//a[text()='Log Out']"));
        System.out.println("Found Log Out link.");

        // Task 03: Indexing
        System.out.println("Executing Task 03: Finding 3rd row in #inventory table...");
        driver.findElement(By.xpath("//table[@id='inventory']//tr[3]"));
        System.out.println("Found 3rd row.");

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
