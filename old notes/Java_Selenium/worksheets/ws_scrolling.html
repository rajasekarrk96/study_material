Task 01: Scroll Down by Pixels
Use JavascriptExecutor with window.scrollBy(0, 500) to scroll the page down by 500 pixels vertically.
Task 02: Scroll to Specific Element
Use scrollIntoView() to scroll until the "Footer" element (id="footer") becomes visible in the viewport.
Task 03: Scroll to Bottom of Page
Use window.scrollTo(0, document.body.scrollHeight) to scroll all the way to the bottom of the page.

package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class _20_Scrolling {
    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");

        WebDriver driver =
        new ChromeDriver();

        System.out.println("Navigating to Scrolling worksheet...");
        driver.get(
            "http://127.0.0.1:5501/Java_Selinium/worksheets/ws_scrolling.html"
        );

        JavascriptExecutor js =
        (JavascriptExecutor) driver;

        // Task 01: Scroll Down by Pixels (500px)
        System.out.println("Executing Task 01: Scrolling down by 500 pixels...");
        js.executeScript(
            "window.scrollBy(0, 500)"
        );

        // Task 02: Scroll to Specific Element
        System.out.println("Executing Task 02: Scrolling 'Footer' into view...");
        WebElement footer =
        driver.findElement(
            By.id("footer")
        );

        js.executeScript(
            "arguments[0].scrollIntoView(true);",
            footer
        );

        // Task 03: Scroll to Bottom of Page
        System.out.println("Executing Task 03: Scrolling to the bottom of the page...");
        js.executeScript(
            "window.scrollTo(0, document.body.scrollHeight)"
        );

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
