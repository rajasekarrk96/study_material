String Comparisons in Java

Different ways to compare strings in Java.

equals() Method

Compares the content of two strings. Returns true if content is same.

public class EqualsDemo {
    public static void main(String[] args) {
        String s1 = "Hello";
        String s2 = "Hello";
        String s3 = new String("Hello");
        
        System.out.println(s1.equals(s2));  // true
        System.out.println(s1.equals(s3));  // true
        System.out.println(s1 == s2);  // true (same reference)
        System.out.println(s1 == s3);  // false (different reference)
    }
}

compareTo() Method

Compares strings lexicographically. Returns 0 if equal, negative if less, positive if greater.

public class CompareToDemo {
    public static void main(String[] args) {
        String s1 = "Apple";
        String s2 = "Banana";
        String s3 = "Apple";
        
        System.out.println(s1.compareTo(s2));  // Negative (Apple < Banana)
        System.out.println(s2.compareTo(s1));  // Positive (Banana > Apple)
        System.out.println(s1.compareTo(s3));  // 0 (equal)
    }
}

equalsIgnoreCase() Method

Compares strings ignoring case differences.

public class IgnoreCaseDemo {
    public static void main(String[] args) {
        String s1 = "Hello";
        String s2 = "HELLO";
        String s3 = "hello";
        
        System.out.println(s1.equals(s2));  // false
        System.out.println(s1.equalsIgnoreCase(s2));  // true
        System.out.println(s1.equalsIgnoreCase(s3));  // true
    }
}
