Regex Quantifiers

Regex Quantifiers

Specify how many times a pattern should match.

Quantifier Types

Quantifier Meaning Example
* 0 or more a* matches "", "a", "aa"
+ 1 or more a+ matches "a", "aa"
? 0 or 1 a? matches "", "a"
{n} Exactly n a{3} matches "aaa"
{n,} n or more a{2,} matches "aa", "aaa"
{n,m} Between n and m a{2,4} matches "aa", "aaa"

Example: Phone Number Validation

import java.util.regex.*;

public class QuantifierDemo {
    public static void main(String[] args) {
        // Pattern: 3 digits, dash, 3 digits, dash, 4 digits
        String pattern = "\\d{3}-\\d{3}-\\d{4}";
        
        String[] phones = {
            "123-456-7890",  // Valid
            "12-456-7890",   // Invalid
            "123-45-7890"    // Invalid
        };
        
        Pattern p = Pattern.compile(pattern);
        for (String phone : phones) {
            Matcher m = p.matcher(phone);
            System.out.println(phone + ": " + m.matches());
        }
    }
}
123-456-7890: true
12-456-7890: false
123-45-7890: false
