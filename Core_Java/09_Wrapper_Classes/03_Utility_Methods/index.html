Wrapper Class Utility Methods

Use built-in methods for parsing, comparison, and conversion.

Parsing Methods

public class ParsingDemo {
    public static void main(String[] args) {
        // String to primitive
        int num = Integer.parseInt("123");
        double decimal = Double.parseDouble("3.14");
        boolean flag = Boolean.parseBoolean("true");
        
        System.out.println("Parsed int: " + num);
        System.out.println("Parsed double: " + decimal);
        System.out.println("Parsed boolean: " + flag);
        
        // String to wrapper object
        Integer numObj = Integer.valueOf("456");
        System.out.println("Integer object: " + numObj);
    }
}
Parsed int: 123
Parsed double: 3.14
Parsed boolean: true
Integer object: 456

Comparison Methods

public class ComparisonDemo {
    public static void main(String[] args) {
        Integer a = 10;
        Integer b = 20;
        
        // compare() method
        int result = Integer.compare(a, b);
        System.out.println("Compare result: " + result);  // Negative
        
        // compareTo() method
        System.out.println("a.compareTo(b): " + a.compareTo(b));
        
        // equals() method
        System.out.println("a.equals(b): " + a.equals(b));
    }
}
Compare result: -1
a.compareTo(b): -1
a.equals(b): false

Constants and Limits

public class ConstantsDemo {
    public static void main(String[] args) {
        System.out.println("Integer MAX: " + Integer.MAX_VALUE);
        System.out.println("Integer MIN: " + Integer.MIN_VALUE);
        System.out.println("Double MAX: " + Double.MAX_VALUE);
        System.out.println("Double MIN: " + Double.MIN_VALUE);
        System.out.println("Character MAX: " + (int)Character.MAX_VALUE);
    }
}
