instanceof Operator

instanceof Operator in Java

Test object types and perform safe type casting with the instanceof operator.

What is instanceof?

The instanceof operator is used to test whether an object is an instance of a specific class or implements a specific interface. It returns true if the object is an instance of the specified type, otherwise false.

Syntax:
object instanceof ClassName
Key Points:
  • Returns boolean value (true or false)
  • Used for type checking before casting
  • Prevents ClassCastException
  • Works with classes, interfaces, and inheritance
  • Returns false if object is null

1. Basic Usage

Example 1: Simple Type Checking
public class InstanceofBasic {
    public static void main(String[] args) {
        String str = "Hello";
        Integer num = 100;
        Double decimal = 3.14;
        
        System.out.println("str instanceof String: " + (str instanceof String));
        System.out.println("num instanceof Integer: " + (num instanceof Integer));
        System.out.println("decimal instanceof Double: " + (decimal instanceof Double));
        System.out.println("str instanceof Integer: " + (str instanceof Integer));
    }
}
str instanceof String: true
num instanceof Integer: true
decimal instanceof Double: true
str instanceof Integer: false
Example 2: instanceof with null
public class InstanceofNull {
    public static void main(String[] args) {
        String str = null;
        
        System.out.println("null instanceof String: " + (str instanceof String));
        
        if (str instanceof String) {
            System.out.println("This won't print");
        } else {
            System.out.println("null is not an instance of any class");
        }
    }
}
null instanceof String: false
null is not an instance of any class
💡 Important: instanceof always returns false for null, which prevents NullPointerException.

2. instanceof with Inheritance

Example 1: Class Hierarchy
class Animal {
    void sound() {
        System.out.println("Animal makes sound");
    }
}

class Dog extends Animal {
    void sound() {
        System.out.println("Dog barks");
    }
}

class Cat extends Animal {
    void sound() {
        System.out.println("Cat meows");
    }
}

public class InstanceofInheritance {
    public static void main(String[] args) {
        Animal animal = new Animal();
        Dog dog = new Dog();
        Cat cat = new Cat();
        Animal animalDog = new Dog();  // Upcasting
        
        System.out.println("animal instanceof Animal: " + (animal instanceof Animal));
        System.out.println("dog instanceof Dog: " + (dog instanceof Dog));
        System.out.println("dog instanceof Animal: " + (dog instanceof Animal));
        System.out.println("animalDog instanceof Dog: " + (animalDog instanceof Dog));
        System.out.println("animalDog instanceof Cat: " + (animalDog instanceof Cat));
    }
}
animal instanceof Animal: true
dog instanceof Dog: true
dog instanceof Animal: true
animalDog instanceof Dog: true
animalDog instanceof Cat: false
Example 2: Interface Implementation
interface Flyable {

class Bird implements Flyable {

bird instanceof Flyable: true
plane instanceof Flyable: true
bird instanceof Bird: true
bird instanceof Airplane: false

3. Safe Downcasting with instanceof

Example 1: Preventing ClassCastException
class Shape {
    void draw() {
        System.out.println("Drawing shape");
    }
}

class Circle extends Shape {
    void draw() {
        System.out.println("Drawing circle");
    }
    
    void calculateArea() {
        System.out.println("Calculating circle area");
    }
}

class Rectangle extends Shape {
    void draw() {
        System.out.println("Drawing rectangle");
    }
    
    void calculatePerimeter() {
        System.out.println("Calculating rectangle perimeter");
    }
}

public class SafeDowncasting {
    public static void main(String[] args) {
        Shape shape1 = new Circle();
        Shape shape2 = new Rectangle();
        
        // Safe downcasting with instanceof
        if (shape1 instanceof Circle) {
            Circle circle = (Circle) shape1;
            circle.calculateArea();
        }
        
        if (shape2 instanceof Rectangle) {
            Rectangle rect = (Rectangle) shape2;
            rect.calculatePerimeter();
        }
        
        // Unsafe - would cause ClassCastException without instanceof check
        // Circle c = (Circle) shape2;  // ERROR at runtime
    }
}
Calculating circle area
Calculating rectangle perimeter
Example 2: Type-Specific Operations
class Employee {
    String name;
    
    Employee(String name) {
        this.name = name;
    }
}

class Manager extends Employee {
    int teamSize;
    
    Manager(String name, int teamSize) {
        super(name);
        this.teamSize = teamSize;
    }
    
    void manageTeam() {
        System.out.println(name + " manages " + teamSize + " employees");
    }
}

class Developer extends Employee {
    String language;
    
    Developer(String name, String language) {
        super(name);
        this.language = language;
    }
    
    void code() {
        System.out.println(name + " codes in " + language);
    }
}

public class EmployeeDemo {
    static void processEmployee(Employee emp) {
        System.out.println("Processing: " + emp.name);
        
        if (emp instanceof Manager) {
            Manager mgr = (Manager) emp;
            mgr.manageTeam();
        } else if (emp instanceof Developer) {
            Developer dev = (Developer) emp;
            dev.code();
        }
    }
    
    public static void main(String[] args) {
        Employee e1 = new Manager("Alice", 5);
        Employee e2 = new Developer("Bob", "Java");
        
        processEmployee(e1);
        processEmployee(e2);
    }
}
Processing: Alice
Alice manages 5 employees
Processing: Bob
Bob codes in Java

Practical Examples

Example 1: Payment Processing System
interface Payment {
    void processPayment(double amount);
}

class CreditCard implements Payment {
    String cardNumber;
    
    CreditCard(String cardNumber) {
        this.cardNumber = cardNumber;
    }
    
    public void processPayment(double amount) {
        System.out.println("Processing $" + amount + " via Credit Card");
    }
    
    void validateCard() {
        System.out.println("Validating card: " + cardNumber);
    }
}

class PayPal implements Payment {
    String email;
    
    PayPal(String email) {
        this.email = email;
    }
    
    public void processPayment(double amount) {
        System.out.println("Processing $" + amount + " via PayPal");
    }
    
    void sendReceipt() {
        System.out.println("Receipt sent to: " + email);
    }
}

public class PaymentDemo {
    static void checkout(Payment payment, double amount) {
        payment.processPayment(amount);
        
        if (payment instanceof CreditCard) {
            CreditCard cc = (CreditCard) payment;
            cc.validateCard();
        } else if (payment instanceof PayPal) {
            PayPal pp = (PayPal) payment;
            pp.sendReceipt();
        }
    }
    
    public static void main(String[] args) {
        Payment p1 = new CreditCard("1234-5678-9012-3456");
        Payment p2 = new PayPal("user@example.com");
        
        checkout(p1, 100.50);
        System.out.println();
        checkout(p2, 75.25);
    }
}
Processing $100.5 via Credit Card
Validating card: 1234-5678-9012-3456

Processing $75.25 via PayPal
Receipt sent to: user@example.com
Example 2: Collection Type Checking
import java.util.*;

public class CollectionTypeCheck {
    static void printCollection(Object obj) {
        if (obj instanceof ArrayList) {
            ArrayList list = (ArrayList) obj;
            System.out.println("ArrayList with " + list.size() + " elements");
        } else if (obj instanceof HashSet) {
            HashSet set = (HashSet) obj;
            System.out.println("HashSet with " + set.size() + " elements");
        } else if (obj instanceof HashMap) {
            HashMap map = (HashMap) obj;
            System.out.println("HashMap with " + map.size() + " entries");
        } else {
            System.out.println("Unknown collection type");
        }
    }
    
    public static void main(String[] args) {
        ArrayList list = new ArrayList<>();
        list.add("A");
        list.add("B");
        
        HashSet set = new HashSet<>();
        set.add(1);
        set.add(2);
        set.add(3);
        
        HashMap map = new HashMap<>();
        map.put("One", 1);
        
        printCollection(list);
        printCollection(set);
        printCollection(map);
    }
}
ArrayList with 2 elements
HashSet with 3 elements
HashMap with 1 entries
Best Practices:
Do Don't
✅ Use instanceof before downcasting ❌ Cast without checking type
✅ Check for null separately if needed ❌ Rely on instanceof for null checks
✅ Use polymorphism when possible ❌ Overuse instanceof (code smell)
✅ Handle all possible types ❌ Assume only one type exists
