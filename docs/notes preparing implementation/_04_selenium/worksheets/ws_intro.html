package Selenium;

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class _03_introselenium {

    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver = new ChromeDriver();
 
        System.out.println("Navigating to Intro worksheet...");
        driver.get("http://127.0.0.1:5501/Java_Selinium/worksheets/ws_intro.html");
 
        // Task 01: pom.xml Dependency
        System.out.println("Executing Task 01: Providing Selenium 4.x dependency block...");
        String pomXml =
            "<dependency>\n" +
            "    <groupId>org.seleniumhq.selenium</groupId>\n" +
            "    <artifactId>selenium-java</artifactId>\n" +
            "    <version>4.21.0</version>\n" +
            "</dependency>";
        System.out.println("Add this to your pom.xml:\n" + pomXml);
 
        // Task 02: Setup
        System.out.println("\nExecuting Task 02: Displaying driver setup strategy...");
        System.out.println("Note: Selenium 4.6+ handles drivers automatically using Selenium Manager.");
        System.out.println("For legacy manual setup, use: WebDriverManager.chromedriver().setup();");
 
        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
