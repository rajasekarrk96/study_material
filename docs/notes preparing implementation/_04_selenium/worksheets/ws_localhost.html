Task 01: Open Local Port
Write code to open an app running on port 8080.
Task 02: Open File
Write code to open "index.html" directly from C: drive.
Solutions 
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class LocalExecutionSolution {
    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver = new ChromeDriver();

        // Task 01: Open Local Port
        System.out.println("Executing Task 01: Opening Local Port 8080...");
        driver.get("http://localhost:8080");

        // Task 02: Open HTML File Protocol
        System.out.println("Executing Task 02: Opening Local HTML File...");
        // Note: Use three slashes 'file:///' for Windows paths
        driver.get("file:///C:/Users/User/Desktop/index.html");

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
