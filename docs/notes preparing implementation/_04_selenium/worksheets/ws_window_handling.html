Task 01: Get Parent Window Handle
Use driver.getWindowHandle() to store the parent window handle in a String variable before opening any new windows.
Task 02: Switch to Child Window
Use driver.getWindowHandles() to get all window handles. Iterate through them and switch to the child window using switchTo().window().
Task 03: Close Child and Return to Parent
Close the current child window using driver.close(). Then switch back to the parent window using the stored handle.

package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import java.util.Set;

public class _23_WindowHandling {

    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver = new ChromeDriver();
        driver.manage().window().maximize();

        System.out.println("Navigating to Window Handling worksheet...");
        driver.get("http://127.0.0.1:5501/Java_Selinium/worksheets/ws_window_handling.html");

        // Task 01: Get Parent Window Handle
        System.out.println("Executing Task 01: Retrieving parent window handle...");
        String parentID = driver.getWindowHandle();
        System.out.println("Parent Window ID: " + parentID);

        // Task 02: Open and switch to Child Window
        System.out.println("Executing Task 02: Opening and switching to child window...");
        driver.findElement(By.linkText("Click Here")).click();

        Set<String> allWindows = driver.getWindowHandles();
        for(String windowID : allWindows) {
            if(!windowID.equals(parentID)) {
                driver.switchTo().window(windowID);
                System.out.println("Switched to Child Window: " + windowID);
            }
        }

        // Task 03: Close Child and return to Parent
        System.out.println("Executing Task 03: Closing child and returning to parent window...");
        driver.close();
        driver.switchTo().window(parentID);
        System.out.println("Returned to Parent Window successfully.");

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
