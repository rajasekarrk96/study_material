package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.Keys;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.interactions.Actions;

public class _18_KeyboardActions {
    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver =
        new ChromeDriver();

        System.out.println("Navigating to Keyboard Actions worksheet...");
        driver.get(
            "http://127.0.0.1:5501/Java_Selinium/worksheets/ws_keyboard.html"
        );

        Actions act =
        new Actions(driver);

        // Task 01: Keyboard (Using IDs as per instruction)
        System.out.println("Locating 'box1' element via ID...");
        WebElement box1 =
        driver.findElement(
            By.id("box1")
        );

        // Task 01: Type in UPPERCASE
        System.out.println("Executing Task 01: Typing 'selenium' in UPPERCASE...");
        act.keyDown(Keys.SHIFT)
           .sendKeys(box1, "selenium")
           .keyUp(Keys.SHIFT)
           .perform();

        // Task 02: Copy Text
        System.out.println("Executing Task 02: Copying text from box1 to box2...");
        act.keyDown(Keys.CONTROL)
           .sendKeys("a")
           .sendKeys("c")
           .keyUp(Keys.CONTROL)
           .perform();

        // Task 02: Paste Text (Using ID as per instruction)
        System.out.println("Locating 'box2' element via ID...");
        WebElement box2 =
        driver.findElement(
            By.id("box2")
        );

        act.click(box2)
           .keyDown(Keys.CONTROL)
           .sendKeys("v")
           .keyUp(Keys.CONTROL)
           .perform();

        // Task 03: Press ENTER Key
        System.out.println("Executing Task 03: Typing 'automation' and pressing ENTER...");
        box1.clear();
        box1.sendKeys("automation");
        box1.sendKeys(Keys.ENTER);

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
