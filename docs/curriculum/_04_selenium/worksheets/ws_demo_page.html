package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;
import java.time.Duration;

public class _01_DemoPage {

    public static void main(String[] args) {


        // Modern Selenium (4.6.0+) handles Driver Management automatically.
        // Alternate Way (Legacy/WebDriverManager):
        // WebDriverManager.chromedriver().setup();
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver = new ChromeDriver();
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(10));

        System.out.println("Navigating to Demo Page worksheet...");
        driver.get("http://127.0.0.1:5501/Java_Selinium/worksheets/ws_demo_page.html");

        // Task 01: Click the Demo Button
        System.out.println("Executing Task 01: Clicking Demo Button...");
        wait.until(ExpectedConditions.elementToBeClickable(By.id("demoBtn"))).click();

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
