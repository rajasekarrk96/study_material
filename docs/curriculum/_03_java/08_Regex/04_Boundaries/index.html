Regex Boundaries

Regex Boundaries

Match positions in strings using boundary matchers.

Boundary Types

Boundary Meaning
^ Start of line
$ End of line
\b Word boundary
\B Non-word boundary

Example: Start and End Matching

import java.util.regex.*;

public class BoundaryDemo {
    public static void main(String[] args) {
        String text = "Java Programming";
        
        // ^ matches start
        System.out.println(text.matches("^Java.*"));  // true
        
        // $ matches end
        System.out.println(text.matches(".*Programming$"));  // true
        
        // \b matches word boundary
        Pattern p = Pattern.compile("\\bJava\\b");
        Matcher m = p.matcher("Java is great, JavaScript too");
        while (m.find()) {
            System.out.println("Found at: " + m.start());
        }
    }
}
