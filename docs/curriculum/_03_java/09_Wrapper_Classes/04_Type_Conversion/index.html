Type Conversion with Wrappers

Convert between different data types using wrapper classes.

Converting to String

public class ToStringDemo {
    public static void main(String[] args) {
        int num = 123;
        double decimal = 3.14;
        boolean flag = true;
        
        // Using toString()
        String str1 = Integer.toString(num);
        String str2 = Double.toString(decimal);
        String str3 = Boolean.toString(flag);
        
        System.out.println("String from int: " + str1);
        System.out.println("String from double: " + str2);
        System.out.println("String from boolean: " + str3);
        
        // Using String.valueOf()
        String str4 = String.valueOf(num);
        System.out.println("Using valueOf: " + str4);
    }
}
String from int: 123
String from double: 3.14
String from boolean: true
Using valueOf: 123

Numeric Conversions

public class NumericConversionDemo {
    public static void main(String[] args) {
        Integer num = 100;
        
        // Integer to other types
        double d = num.doubleValue();
        long l = num.longValue();
        byte b = num.byteValue();
        
        System.out.println("Double: " + d);
        System.out.println("Long: " + l);
        System.out.println("Byte: " + b);
        
        // String to different numeric types
        String str = "42";
        int i = Integer.parseInt(str);
        double dbl = Double.parseDouble(str);
        long lng = Long.parseLong(str);
        
        System.out.println("Int: " + i);
        System.out.println("Double: " + dbl);
        System.out.println("Long: " + lng);
    }
}
Double: 100.0
Long: 100
Byte: 100
Int: 42
Double: 42.0
Long: 42

Binary and Hex Conversion

public class BaseConversionDemo {
    public static void main(String[] args) {
        int num = 42;
        
        // To binary
        String binary = Integer.toBinaryString(num);
        System.out.println("Binary: " + binary);
        
        // To hex
        String hex = Integer.toHexString(num);
        System.out.println("Hex: " + hex);
        
        // To octal
        String octal = Integer.toOctalString(num);
        System.out.println("Octal: " + octal);
        
        // From binary to int
        int fromBinary = Integer.parseInt("101010", 2);
        System.out.println("From binary: " + fromBinary);
    }
}
Binary: 101010
Hex: 2a
Octal: 52
From binary: 42
