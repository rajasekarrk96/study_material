import org.openqa.selenium.PageLoadStrategy;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;

public class _08_Pageloadstrategy {
    public static void main(String[] args) {

        System.out.println("--- Selenium Lab Session Started ---");
        ChromeOptions options =
            new ChromeOptions();

        // Task 01: Eager Load (Waits for DOM but not images)
        System.out.println("Executing Task 01: Setting Page Load Strategy to EAGER...");
        options.setPageLoadStrategy(
            PageLoadStrategy.EAGER
        );

        // Task 02: Normal Load (Default strategy - no code needed)
        System.out.println("Note: Task 02 - NORMAL is the default strategy.");

        // Task 03: None (Returns immediately after HTML starts)
        System.out.println("Note: Task 03 - NONE strategy can be set via options.");
        // options.setPageLoadStrategy(PageLoadStrategy.NONE);

        WebDriver driver =
            new ChromeDriver(options);

        // Open Page Load Strategy worksheet
        System.out.println("Opening Page Load Strategy worksheet...");
        driver.get(
            "http://127.0.0.1:5501/Java_Selinium/worksheets/ws_page_load_strategy.html"
        );

        System.out.println(("--- Selenium Lab Session Ended ---"));
        driver.quit();
    }
}
Strategies
NORMAL Waits for entire page (HTML + CSS + Images)
EAGER Waits for HTML only (DOMContentLoaded)
NONE Does not wait at all
