package Selenium;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;
import java.time.Duration;

public class _16_AlertsandPopups {
    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver = new ChromeDriver();
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(10));

        System.out.println("Navigating to Alerts worksheet...");
        driver.get("http://127.0.0.1:5501/Java_Selinium/worksheets/ws_alerts.html");

        // TASK 01: Alert
        System.out.println("Executing Task 01: Handling Alert...");
        driver.findElement(By.xpath("//button[text()='Alert Me']")).click();
        wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("mockAlert")));
        driver.findElement(By.xpath("//div[@id='mockAlert']//button[text()='OK']")).click();
        System.out.println("Alert accepted.");

        // TASK 02: Confirm
        System.out.println("Executing Task 02: Handling Confirm...");
        driver.findElement(By.xpath("//button[text()='Confirm Me']")).click();
        wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("mockConfirm")));
        driver.findElement(By.xpath("//div[@id='mockConfirm']//button[text()='Cancel']")).click();
        System.out.println("Confirm dismissed.");

        // TASK 03: Prompt
        System.out.println("Executing Task 03: Handling Prompt...");
        driver.findElement(By.xpath("//button[text()='Prompt Me']")).click();
        wait.until(ExpectedConditions.visibilityOfElementLocated(By.id("mockPrompt")));
        driver.findElement(By.id("promptInput")).sendKeys("Admin");
        driver.findElement(By.xpath("//div[@id='mockPrompt']//button[text()='OK']")).click();
        System.out.println("Prompt handled with input: Admin");

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
