import com.aventstack.extentreports.ExtentReports;
import com.aventstack.extentreports.ExtentTest;
import com.aventstack.extentreports.reporter.ExtentSparkReporter;

public class ReportingSolution {
    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");

        // Task 01: Setup
        System.out.println("Executing Task 01: Setting up ExtentReports and SparkReporter...");
        ExtentReports extent = new ExtentReports();
        ExtentSparkReporter spark = new ExtentSparkReporter("report.html");
        extent.attachReporter(spark);

        // Task 02: Log Status
        System.out.println("Executing Task 02: Creating test and logging PASS status...");
        ExtentTest test = extent.createTest("Login Test");
        test.pass("Login Successful");

        // Final Step: Flush
        System.out.println("Flushing report to 'report.html'...");
        extent.flush();

        System.out.println("--- Selenium Lab Session Ended ---");
    }
}
