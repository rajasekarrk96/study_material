Task 01: Select by Visible Text
Select "India" from the Country dropdown using selectByVisibleText(). Print the selected option text to confirm.
(Instruction: Use the ID locator 'country' for this task)
Task 02: Select by Index
Select the second option (Index 2) from the dropdown using selectByIndex(). Verify which country was selected.
Task 03: Get All Options
Use getOptions() to retrieve all dropdown options. Print the total count and all option texts in a loop.

package Selenium;

import java.util.List;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.Select;

public class _13_DropDownHandling {
    public static void main(String[] args) {
        System.out.println("--- Selenium Lab Session Started ---");
        WebDriver driver = new ChromeDriver();

        System.out.println("Navigating to Dropdown Handling worksheet...");
        driver.get("http://127.0.0.1:5501/Java_Selinium/worksheets/ws_dropdown.html");

        // Task 01: Select by Visible Text
        System.out.println("Executing Task 01: Selecting India by visible text...");
        WebElement dd = driver.findElement(By.id("country"));
        Select select = new Select(dd);
        select.selectByVisibleText("India");
        System.out.println("Selected Option: " + select.getFirstSelectedOption().getText());

        // Task 02: Select by Index
        System.out.println("Executing Task 02: Selecting option at Index 2...");
        select.selectByIndex(2);
        System.out.println("Selected Option: " + select.getFirstSelectedOption().getText());

        // Task 03: Get All Options
        System.out.println("Executing Task 03: Retrieving all dropdown options...");
        List<WebElement> allOptions = select.getOptions();
        System.out.println("Total options found: " + allOptions.size());

        for(WebElement option : allOptions) {
            System.out.println("- " + option.getText());
        }

        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
