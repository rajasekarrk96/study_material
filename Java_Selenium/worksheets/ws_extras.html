Task 02: Command Line Params
How do you pass the browser name from Maven command like `mvn test -Dbrowser=...`?
Solutions 

package Selenium;

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;

public class _31_Extras {

    public static void main
    (String[] args)
    {

        System.out.println("--- Selenium Lab Session Started ---");
 
        // Task 01: Headless Execution
        System.out.println("Executing Task 01: Configuring and launching Chrome in headless mode...");
        ChromeOptions options = new ChromeOptions();
        options.addArguments("--headless");
        WebDriver driver = new ChromeDriver(options);
 
        System.out.println("Navigating to Extras worksheet...");
        driver.get("http://127.0.0.1:5501/Java_Selinium/worksheets/ws_extras.html");
 
        // Task 02: Dynamic Browser Selection
        System.out.println("Executing Task 02: Retrieving dynamic browser property...");
        String browser = System.getProperty("browser", "chrome");
        System.out.println("Selected Browser: " + browser);
 
        System.out.println("--- Selenium Lab Session Ended ---");
        driver.quit();
    }
}
