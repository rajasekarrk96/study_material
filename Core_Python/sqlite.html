Python SQLite

Introduction

Python has built-in support for SQLite3, which provides an interface for interacting with SQLite databases.

To use it, you must first create a connection variable that connects to the database.

Create Connection

Syntax:

conn = sqlite3.connect(database_name)

Example:

import sqlite3

# This will create 'test.db' if it doesn't exist
conn = sqlite3.connect('test.db')
print("Opened database successfully")

Create Table

To execute SQL statements (like CREATE TABLE), you need a cursor object.

conn = sqlite3.connect('test.db')
print("Opened database successfully")

conn.execute('''CREATE TABLE COMPANY
         (ID INT PRIMARY KEY     NOT NULL,
         NAME           TEXT    NOT NULL,
         AGE            INT     NOT NULL,
         ADDRESS        CHAR(50),
         SALARY         REAL);''')

print("Table created successfully")
conn.close()

Insert Operation

To insert records into a table, use the INSERT statement.

conn = sqlite3.connect('test.db')

conn.execute("INSERT INTO COMPANY (ID,NAME,AGE,ADDRESS,SALARY) \
      VALUES (1, 'Paul', 32, 'California', 20000.00 )")

conn.execute("INSERT INTO COMPANY (ID,NAME,AGE,ADDRESS,SALARY) \
      VALUES (2, 'Allen', 25, 'Texas', 15000.00 )")

conn.execute("INSERT INTO COMPANY (ID,NAME,AGE,ADDRESS,SALARY) \
      VALUES (3, 'Teddy', 23, 'Norway', 20000.00 )")

conn.commit() # Save changes
print("Records created successfully")
conn.close()

Select Operation

To retrieve data, use the SELECT statement and fetch the results.

conn = sqlite3.connect('test.db')
cursor = conn.execute("SELECT id, name, address, salary from COMPANY")

for row in cursor:
   print("ID = ", row[0])
   print("NAME = ", row[1])
   print("ADDRESS = ", row[2])
   print("SALARY = ", row[3], "\n")

print("Operation done successfully")
conn.close()

Update Operation

Use the UPDATE statement to modify existing records.

conn = sqlite3.connect('test.db')

conn.execute("UPDATE COMPANY set SALARY = 25000.00 where ID = 1")
conn.commit()

print("Total number of rows updated :", conn.total_changes)

cursor = conn.execute("SELECT id, name, address, salary from COMPANY")
for row in cursor:
   print("ID = ", row[0])
   print("NAME = ", row[1])
   print("SALARY = ", row[3], "\n")

conn.close()

Delete Operation

Use the DELETE statement to remove records.

conn = sqlite3.connect('test.db')

conn.execute("DELETE from COMPANY where ID = 2;")
conn.commit()

print("Total number of rows deleted :", conn.total_changes)

cursor = conn.execute("SELECT id, name, address, salary from COMPANY")
for row in cursor:
   print("ID = ", row[0])
   print("NAME = ", row[1])
   print("SALARY = ", row[3], "\n")

conn.close()

Worksheet

1. Write a program to connect to an SQLite database named 'school.db'.

import sqlite3
conn = sqlite3.connect('school.db')
print("Connected")
conn.close()

2. Write a program to create a table 'STUDENTS' with ID, Name, and Grade.

import sqlite3
conn = sqlite3.connect('school.db')
conn.execute('''CREATE TABLE IF NOT EXISTS STUDENTS
         (ID INT PRIMARY KEY NOT NULL,
         NAME TEXT NOT NULL,
         GRADE TEXT NOT NULL);''')
print("Table created")
conn.close()

3. Write a program to insert a student into the STUDENTS table.

import sqlite3
conn = sqlite3.connect('school.db')
conn.execute("INSERT INTO STUDENTS (ID,NAME,GRADE) VALUES (1, 'Alice', 'A')")
conn.commit()
print("Inserted")
conn.close()

4. Write a program to select and print all students from the table.

import sqlite3
conn = sqlite3.connect('school.db')
cursor = conn.execute("SELECT * from STUDENTS")
for row in cursor:
    print(row)
conn.close()

5. Write a program to update a student's grade.

import sqlite3
conn = sqlite3.connect('school.db')
conn.execute("UPDATE STUDENTS set GRADE = 'A+' where ID = 1")
conn.commit()
print("Updated")
conn.close()

6. Write a program to delete a student from the table using ID.

import sqlite3
conn = sqlite3.connect('school.db')
conn.execute("DELETE from STUDENTS where ID = 1")
conn.commit()
print("Deleted")
conn.close()

7. Write a program to insert multiple rows using excutemany.

import sqlite3
conn = sqlite3.connect('school.db')
data = [(2, 'Bob', 'B'), (3, 'Charlie', 'C')]
conn.executemany("INSERT INTO STUDENTS (ID, NAME, GRADE) VALUES (?, ?, ?)", data)
conn.commit()
print("Inserted Many")
conn.close()

8. Write a program to fetch only one row using fetchone().

import sqlite3
conn = sqlite3.connect('school.db')
cursor = conn.execute("SELECT * from STUDENTS")
print(cursor.fetchone())
conn.close()

9. Write a program to drop the STUDENTS table.

import sqlite3
conn = sqlite3.connect('school.db')
conn.execute("DROP TABLE IF EXISTS STUDENTS")
print("Table Dropped")
conn.close()

10. Write a program to handle database connection errors using try-except.

import sqlite3
try:
    conn = sqlite3.connect('school.db')
    print("Success")
except sqlite3.Error as e:
    print("Error:", e)
finally:
    if conn:
        conn.close()
